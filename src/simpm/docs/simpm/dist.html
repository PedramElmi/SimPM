 12def fit(data,dist_type,method='mle'):
 13    '''
 14    Fit a distribution on data
 15
 16    Parameters:
 17    -----------
 18    data : array_like
 19        The data to fit the distribution to.
 20    dist_type : str
 21        The type of distribution to fit. Supported distribution types are 'triang', 'norm', 'beta', 'trapz', and 'expon'.
 22    method : str, optional
 23        The method to use for fitting the distribution. Default is 'mle' (maximum likelihood estimation).
 24
 25    Returns:
 26    --------
 27    fitted_distribution : object
 28        An object representing the fitted distribution.
 29
 30    Raises:
 31    -------
 32    ValueError
 33        If the distribution type is not recognized.
 34    '''
 35
 36    try:
 37        data=np.concatenate(data).ravel()
 38    except:
 39        pass
 40    a=None
 41    if dist_type=='triang':
 42        dist_type='triang'
 43        params=st.triang.fit(data)
 44        dist=st.triang(params[0],loc=params[1],scale=params[2])
 45        a=triang(0,1,2)
 46        a.dist=dist
 47        
 48    elif dist_type=='norm' :
 49        dist_type='norm'
 50        params=st.norm.fit(data,method=method)
 51        dist=st.norm(loc=params[0],scale=params[1])
 52        a=norm(0,1)
 53        a.dist=dist
 54       
 55    elif dist_type=='beta' :
 56            dist_type='beta'
 57            params=st.beta.fit(data,method=method)
 58            dist=st.beta(params[0],params[1],loc=params[2],scale=params[3])
 59            a=beta(1,1,0,1)
 60            a.dist=dist
 61            
 62    elif dist_type=='trapz' :
 63            dist_type='trapz'
 64            params=st.trapz.fit(data,method=method)
 65            
 66            dist=st.trapz(params[0],params[1],loc=params[2],scale=params[3])
 67            a=trapz(1,2,3,4)
 68            a.dist=dist
 69    elif dist_type == 'expon':
 70        params = st.expon.fit(data)
 71        dist = st.expon(scale=params[0])
 72        a = expon(0)
 73        a.dist = dist
 74    else:
 75        raise("Distribution type is not recognized!")
 76        return
 77    a.ks=st.kstest(data,a.dist.cdf)[0]
 78    return a

582    def pdf(self, x):
583        '''
584        Evaluates the probability density function (PDF) of the empirical distribution at the given point.
585
586        Parameters:
587        -----------
588        x : float
589            The value at which to evaluate the PDF.
590
591        Returns:
592        --------
593        float
594            The PDF value at the given point.
595        '''
596        bins = int(2 * len(self.data) ** (1/3))
597        value, BinList = np.histogram(self.data, bins)
598        if x < BinList[0] or x > BinList[-1]:
599            return 0
600        i = 0
601        bl = len(BinList)
602        while x >= BinList[i] and i < bl:
603            i += 1
604        r = value[i-1] / len(self.data)
605        l = BinList[-1] - BinList[0]
606        n = len(BinList)
607        width = l / n
608        r = r / width
609        return r
